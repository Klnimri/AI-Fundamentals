% Define a predicate to check if a language is supported
is_supported_language(Lang) :-
    language(Lang).

% Define a predicate to perform the translation
translate(FromLang, FromPhrase, ToLang, ToPhrase) :-
    is_supported_language(FromLang),
    is_supported_language(ToLang),
    translate_helper(FromLang, FromPhrase, ToLang, ToPhrase).

% Define a helper predicate to find the translation
translate_helper(FromLang, FromPhrase, ToLang, ToPhrase) :-
    translate(FromLang, FromPhrase, ToLang, ToPhrase).

% Define a predicate to handle errors
translate(_, _, _, _) :-
    write("Error: Unsupported source or target language or translation not found").
